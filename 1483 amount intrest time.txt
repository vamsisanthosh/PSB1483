//calculate simple and compound intrest for various parameter specified by the user
#include<stdio.h>
int main()
{
    float p,t,r,si;
    printf("enter the principle amount");
    scanf("%f",&p);
    printf("enter the time in years:");
    scanf("%f",&t);
    printf("enter the rate of intrest:");
    scanf("%f",&r);
    si=(p*t*r)/100;
    printf("simple intrest id %.2f",si);
    
    
}
